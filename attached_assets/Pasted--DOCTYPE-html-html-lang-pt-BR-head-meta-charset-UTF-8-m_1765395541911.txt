<script>
    let currentOption = 1;
    const totalOptions = 5;
    const intervalTime = 20000; // 20 segundos
    let autoSlideInterval;

    // Função para ativar uma opção específica
    function setOption(id) {
        // Atualiza o estado interno
        currentOption = id;

        // Manipulação do DOM (troca as classes)
        document.querySelectorAll('.visual-option').forEach(el => el.classList.remove('active'));
        document.querySelectorAll('.nav-btn').forEach(el => el.classList.remove('active'));
        document.getElementById('opt-' + id).classList.add('active');
        document.getElementById('btn-' + id).classList.add('active');

        // Reinicia o cronômetro se houver interação manual
        clearInterval(autoSlideInterval);
        startAutoSlide();
    }

    // Função para ir para o próximo slide automaticamente
    function nextOption() {
        currentOption++;
        if (currentOption > totalOptions) {
            currentOption = 1;
        }
        setOption(currentOption);
    }

    // Inicia o intervalo
    function startAutoSlide() {
        autoSlideInterval = setInterval(nextOption, intervalTime);
    }

    // Inicializa ao carregar
    startAutoSlide();
</script>
